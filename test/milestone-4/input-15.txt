program testVarForwarding;
var x, y: integer;

procedure increment(var a: integer);
begin
    a := a + 1
end;

procedure addFive(var b: integer);
begin
    b := b + 5
end;

procedure doubleIt(var c: integer);
begin
    c := c * 2
end;

procedure applyAll(var val: integer);
begin
    increment(val);
    addFive(val);
    doubleIt(val)
end;

begin
    x := 1;
    applyAll(x);
    writeln(x);

    y := 10;
    applyAll(y);
    writeln(y)
end.
